# Build stage
FROM golang:1.22-alpine AS builder

RUN apk add --no-cache gcc musl-dev

WORKDIR /app

# 1. Copiar definições de módulos
COPY go.mod ./

# 2. Copiar todo o código-fonte (necessário para que o go mod tidy saiba o que deve baixar/limpar)
COPY . .

# 3. Agora sim, realizar a análise semântica e resolver dependências externas (como argon2)
RUN go mod tidy

# Build main binary
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w" -o cascata-crypto-engine main.go

# Build a tiny healthcheck binary in Go (Enterprise way for distroless)
RUN echo 'package main; import ("net/http"; "os"); func main() { resp, err := http.Get("http://localhost:3000/v1/health"); if err != nil || resp.StatusCode != 200 { os.Exit(1) } }' > healthcheck.go
RUN CGO_ENABLED=0 GOOS=linux go build -o /cce-healthcheck healthcheck.go

# Production stage
FROM gcr.io/distroless/static:nonroot

WORKDIR /

COPY --from=builder /app/cascata-crypto-engine /cascata-crypto-engine
COPY --from=builder /cce-healthcheck /cce-healthcheck

# Porta padrão interna
EXPOSE 3000

USER 65532:65532

ENTRYPOINT ["/cascata-crypto-engine"]
