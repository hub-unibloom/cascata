# Build stage
FROM golang:1.22-alpine AS builder

# Instala bibliotecas necessárias para compilação (se houver CGO, mas não estamos usando)
RUN apk add --no-cache gcc musl-dev

WORKDIR /app

# Primeiro copia apenas go.mod/sum para aproveitar o cache do Docker
COPY go.mod ./
# Se houver go.sum, copie também. Se não, docker build falha se omitido mas arquivo presente.
# COPY go.sum ./

RUN go mod download

# Copia código fonte
COPY . .

# Compila o binário estático (sem dependências de runtime)
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w" -o cascata-crypto-engine main.go

# Production stage
# Usamos distroless/static para máxima segurança (sem shell, sem pacotes, ideal p/ ambientes imutáveis)
FROM gcr.io/distroless/static:nonroot

WORKDIR /

COPY --from=builder /app/cascata-crypto-engine /cascata-crypto-engine

EXPOSE 50051

# Executa como usuário non-root já embutido na imagem
USER 65532:65532

ENTRYPOINT ["/cascata-crypto-engine"]
