
server {
    listen 80;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    # Aumentar tamanho do body para uploads via Dashboard
    client_max_body_size 100M;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # CRITICAL: Prevent Nginx and Cloudflare from caching the Service Worker
    # If cached, browsers will never receive security updates
    location = /service-worker.js {
        expires off;
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0";
    }

    # Cache de assets est√°ticos do React
    location /assets/ {
        expires 1y;
        add_header Cache-Control "public, no-transform";
    }
}
