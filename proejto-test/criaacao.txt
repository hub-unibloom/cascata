-- 1. Garantia de Schema e Helpers
CREATE SCHEMA IF NOT EXISTS auth;

CREATE OR REPLACE FUNCTION auth.uid() RETURNS uuid AS $$
  SELECT NULLIF(current_setting('request.jwt.claim.sub', true), '')::uuid;
$$ LANGUAGE sql STABLE;

CREATE OR REPLACE FUNCTION auth.role() RETURNS text AS $$
  SELECT NULLIF(current_setting('request.jwt.claim.role', true), '')::text;
$$ LANGUAGE sql STABLE;

-- 2. SCHEMA CONCATENATION: PERFIS DISTINTOS
-- A voz é universal, mas os perfis trazem responsabilidades diferentes.

CREATE TABLE IF NOT EXISTS public.politicos (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    nome_publico TEXT NOT NULL,
    partido TEXT,
    cargo_atual TEXT,
    biografia TEXT,
    budget_total INTEGER DEFAULT 1000, -- Políticos também têm budget (voz ativa)
    redes_sociais JSONB DEFAULT '{}',
    verificado BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS public.eleitores (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    apelido TEXT,
    cidade TEXT,
    estado TEXT,
    budget_total INTEGER DEFAULT 500, -- Cidadãos começam com 500 pontos
    interesses_tags TEXT[],
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);

-- 3. PROJETOS DE LEI E INICIATIVAS
CREATE TABLE IF NOT EXISTS public.projetos_lei (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    politico_autor_id UUID REFERENCES public.politicos(id),
    titulo TEXT NOT NULL,
    descricao TEXT NOT NULL,
    escala TEXT NOT NULL CHECK (escala IN ('bairro', 'cidade', 'estado', 'nacional')),
    localidade TEXT,
    status TEXT DEFAULT 'em_analise',
    created_at TIMESTAMPTZ DEFAULT now()
);

-- 4. PARTICIPAÇÃO UNIVERSAL (Referência a auth.users para incluir todos)
-- Aqui garantimos que ninguém seja calado.
CREATE TABLE IF NOT EXISTS public.votos (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    projeto_id UUID NOT NULL REFERENCES public.projetos_lei(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, -- Referência Universal
    voto INTEGER NOT NULL CHECK (voto IN (1, -1)),
    UNIQUE(projeto_id, user_id),
    created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS public.budget_apoio (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    projeto_id UUID NOT NULL REFERENCES public.projetos_lei(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, -- Referência Universal
    pontos_alocados INTEGER NOT NULL CHECK (pontos_alocados >= 1 AND pontos_alocados <= 100),
    UNIQUE(projeto_id, user_id),
    created_at TIMESTAMPTZ DEFAULT now()
);

-- =============================================================================
-- MOTOR DE INTEGRIDADE (Budget Universal)
-- =============================================================================

CREATE OR REPLACE FUNCTION public.validar_gasto_budget()
RETURNS TRIGGER AS $$
DECLARE
    v_budget_disponivel INTEGER;
    v_ja_gasto INTEGER;
BEGIN
    -- Busca o budget do usuário, seja ele Político ou Eleitor
    SELECT COALESCE(
        (SELECT budget_total FROM public.eleitores WHERE id = NEW.user_id),
        (SELECT budget_total FROM public.politicos WHERE id = NEW.user_id)
    ) INTO v_budget_disponivel;
    
    IF v_budget_disponivel IS NULL THEN
        RAISE EXCEPTION 'Perfil não encontrado. Crie seu perfil antes de participar das decisões.';
    END IF;

    SELECT COALESCE(SUM(pontos_alocados), 0) INTO v_ja_gasto 
    FROM public.budget_apoio 
    WHERE user_id = NEW.user_id AND id != NEW.id;

    IF (v_ja_gasto + NEW.pontos_alocados) > v_budget_disponivel THEN
        RAISE EXCEPTION 'Saldo insuficiente. Você possui % pontos de influência livres.', (v_budget_disponivel - v_ja_gasto);
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS check_budget_limit ON public.budget_apoio;
CREATE TRIGGER check_budget_limit BEFORE INSERT OR UPDATE ON public.budget_apoio FOR EACH ROW EXECUTE FUNCTION public.validar_gasto_budget();

-- Visibilidade Pública para Projetos e Políticos
CREATE POLICY "Projetos: Visão Global" ON public.projetos_lei FOR SELECT USING (true);
CREATE POLICY "Politicos: Visão Global" ON public.politicos FOR SELECT USING (true);

-- Privacidade e Controle de Voto
CREATE POLICY "Eleitores: Controle Individual" ON public.eleitores FOR ALL USING (auth.uid() = id);
CREATE POLICY "Politicos: Controle Individual" ON public.politicos FOR ALL USING (auth.uid() = id);

-- Voto Universal: Se está autenticado e tem perfil, tem voz!
CREATE POLICY "Votos: Voz Ativa" ON public.votos FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Budget: Voz Ativa" ON public.budget_apoio FOR ALL USING (auth.uid() = user_id);
