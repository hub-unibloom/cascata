# Cascata Phantom Injection - Tier 1 Core Image
# Base: PostgreSQL 17 Alpine + pgvector (180MB)
FROM postgres:17-alpine AS builder

# Install build dependencies for pgvector
RUN apk add --no-cache \
    build-base \
    clang15 \
    llvm15 \
    git \
    coreutils

# Download and compile pgvector (Tier 1 core requirement for AI)
RUN mkdir -p /tmp/build && \
    cd /tmp/build && \
    git clone --branch v0.8.0 https://github.com/pgvector/pgvector.git && \
    cd pgvector && \
    make && \
    make install

# Final Stage: Lean Alpine Image
FROM postgres:17-alpine

# Copy pgvector from builder
COPY --from=builder /usr/local/share/postgresql/extension/vector* /usr/local/share/postgresql/extension/
COPY --from=builder /usr/local/lib/postgresql/vector.so /usr/local/lib/postgresql/

# Copy the Phantom Linker
COPY phantom_linker.sh /usr/local/bin/phantom_linker
RUN chmod +x /usr/local/bin/phantom_linker

# Custom Entrypoint Wrapper
COPY docker-entrypoint-wrapper.sh /usr/local/bin/
RUN chmod +x /usr/local/bin/docker-entrypoint-wrapper.sh

# The main extension payload volume
VOLUME ["/cascata_extensions"]

ENTRYPOINT ["docker-entrypoint-wrapper.sh"]
CMD ["postgres", "-c", "listen_addresses=*"]
